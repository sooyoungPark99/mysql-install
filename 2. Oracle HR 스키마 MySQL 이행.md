컬럼, 제약조건, 시퀀스, 뷰, 트리거를 확인한다.

```sql
-- 컬럼 정보
COL table_name  FOR A15
COL column_name FOR A25
COL data_type   FOR A15
SET LINESIZE 120

SELECT table_name, column_name, data_type,
       data_precision, data_scale, char_length, nullable
FROM   all_tab_columns
WHERE  owner = 'HR'
ORDER  BY table_name, column_id;

-- 제약 조건
COL table_name        FOR A15
COL constraint_name   FOR A25
COL constraint_type   FOR A5
COL r_constraint_name FOR A25

SELECT table_name, constraint_name, constraint_type, r_constraint_name
FROM   all_constraints
WHERE  owner = 'HR'
  AND  constraint_type IN ('P','U','R')
ORDER  BY table_name, constraint_type;

-- 시퀀스
COL sequence_name FOR A30

SELECT sequence_name, last_number
FROM   all_sequences
WHERE  sequence_owner = 'HR';

-- 뷰
COL view_name FOR A30
SELECT view_name FROM all_views WHERE owner = 'HR';

-- 트리거
COL trigger_name FOR A30
COL status       FOR A10
SELECT trigger_name, status FROM all_triggers WHERE owner = 'HR';

EXIT;
```

## 6. Oracle → MySQL DDL 자동 생성 (oracle)

### 6-1. DDL 생성 스크립트 작성

```bash
vi ~/00_generate_ddl.py
```

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
import cx_Oracle

ORA_USER = "hr"
ORA_PASS = "hr"
ORA_DSN  = "172.31.0.250:1521/orcl"
SCHEMA   = "HR"
OUT_FILE = "/home/oracle/01_hr_mysql_ddl.sql"

def map_type(data_type, precision, scale, char_length):
    if data_type == "VARCHAR2":
        return "VARCHAR({})".format(char_length)
    elif data_type == "CHAR":
        return "CHAR({})".format(char_length)
    elif data_type == "DATE":
        return "DATETIME"
    elif data_type == "NUMBER":
        if precision is None:
            return "DECIMAL(38,10)"
        elif scale is None or scale == 0:
            if precision <= 4:
                return "SMALLINT"
            elif precision <= 9:
                return "INT"
            else:
                return "BIGINT"
        else:
            return "DECIMAL({},{})".format(precision, scale)
    elif data_type in ("CLOB", "NCLOB"):
        return "LONGTEXT"
    elif data_type in ("BLOB", "RAW"):
        return "LONGBLOB"
    else:
        return "TEXT"

def main():
    con = cx_Oracle.connect(ORA_USER, ORA_PASS, ORA_DSN)
    cur = con.cursor()
    print("[OK] Oracle 접속 성공")

    lines = []
    lines.append("SET FOREIGN_KEY_CHECKS = 0;")
    lines.append("")

    cur.execute("SELECT table_name FROM all_tables WHERE owner = :s ORDER BY table_name", s=SCHEMA)
    tables = [r[0] for r in cur.fetchall()]
    print("[INFO] 테이블 {}개 발견".format(len(tables)))

    # PK 수집
    cur.execute("""
        SELECT cc.table_name, cc.column_name
        FROM all_cons_columns cc
        JOIN all_constraints c ON cc.owner=c.owner AND cc.constraint_name=c.constraint_name
        WHERE c.owner=:s AND c.constraint_type='P'
        ORDER BY cc.table_name, cc.position
    """, s=SCHEMA)
    pk_map = {}
    for tbl, col in cur.fetchall():
        pk_map.setdefault(tbl, []).append(col)

    for tbl in tables:
        cur.execute("""
            SELECT column_name, data_type, data_precision,
                   data_scale, char_length, nullable
            FROM all_tab_columns
            WHERE owner=:s AND table_name=:t
            ORDER BY column_id
        """, s=SCHEMA, t=tbl)
        cols = cur.fetchall()

        lines.append("DROP TABLE IF EXISTS {};".format(tbl.lower()))
        lines.append("CREATE TABLE {} (".format(tbl.lower()))
        col_defs = []
        for cname, dtype, prec, scale, clen, nullable in cols:
            mysql_type = map_type(dtype, prec, scale, clen)
            null_str = "" if nullable == "Y" else " NOT NULL"
            col_defs.append("  {} {}{}".format(cname.lower(), mysql_type, null_str))

        if tbl in pk_map:
            pk_cols = ", ".join(c.lower() for c in pk_map[tbl])
            col_defs.append("  PRIMARY KEY ({})".format(pk_cols))

        lines.append(",\n".join(col_defs))
        lines.append(") ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;")
        lines.append("")

    lines.append("SET FOREIGN_KEY_CHECKS = 1;")
    lines.append("")

    cur.close()
    con.close()

    with open(OUT_FILE, "w") as f:
        f.write("\n".join(lines))

    print("[완료] {} 생성됨".format(OUT_FILE))

if __name__ == "__main__":
    main()
```

## 9. 데이터 이행 스크립트 실행 (oracle)

```bash
su - oracle
vi ~/02_hr_migrate_data.py
```

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
import cx_Oracle
import mysql.connector
import sys

ORA_USER = "hr"
ORA_PASS = "hr"
ORA_DSN  = "172.31.0.250:1521/orcl"

MY_HOST  = "127.0.0.1"
MY_PORT  = 3306
MY_USER  = "root"
MY_PASS  = "실제_비밀번호_입력"   # ← 수정 필요
MY_DB    = "hr"

# FK 의존성 순서 (부모 → 자식)
TABLE_ORDER = [
    "REGIONS", "COUNTRIES", "LOCATIONS",
    "JOBS", "DEPARTMENTS", "EMPLOYEES", "JOB_HISTORY",
]

def get_columns(ora_cur, table):
    ora_cur.execute("""
        SELECT column_name FROM user_tab_columns
        WHERE table_name = :t ORDER BY column_id
    """, t=table)
    return [row[0] for row in ora_cur.fetchall()]

def migrate_table(ora_cur, my_cur, table):
    cols = get_columns(ora_cur, table)
    col_list = ", ".join(cols)
    placeholders = ", ".join(["%s"] * len(cols))
    my_table = table.lower()

    ora_cur.execute("SELECT {} FROM {}".format(col_list, table))
    rows = ora_cur.fetchall()

    if not rows:
        print("[SKIP] {} - 데이터 없음".format(table))
        return 0

    sql = "INSERT INTO {} ({}) VALUES ({})".format(
        my_table, col_list.lower(), placeholders
    )
    my_cur.executemany(sql, rows)
    print("[OK] {} - {}건 이행 완료".format(table, len(rows)))
    return len(rows)

def main():
    try:
        ora_con = cx_Oracle.connect(ORA_USER, ORA_PASS, ORA_DSN)
        ora_cur = ora_con.cursor()
        print("[OK] Oracle 접속 성공")
    except Exception as e:
        print("[ERROR] Oracle 접속 실패:", e)
        sys.exit(1)

    try:
        my_con = mysql.connector.connect(
            host=MY_HOST, port=MY_PORT,
            user=MY_USER, password=MY_PASS, database=MY_DB
        )
        my_cur = my_con.cursor()
        print("[OK] MySQL 접속 성공")
    except Exception as e:
        print("[ERROR] MySQL 접속 실패:", e)
        sys.exit(1)

    my_cur.execute("SET FOREIGN_KEY_CHECKS = 0")

    total = 0
    for table in TABLE_ORDER:
        try:
            total += migrate_table(ora_cur, my_cur, table)
        except Exception as e:
            print("[ERROR] {} 이행 실패: {}".format(table, e))

    my_cur.execute("SET FOREIGN_KEY_CHECKS = 1")
    my_con.commit()
    print("\n[완료] 총 {}건 이행".format(total))

    ora_cur.close(); ora_con.close()
    my_cur.close();  my_con.close()

if __name__ == "__main__":
    main()
```

> `MY_PASS` 부분을 실제 MySQL root 비밀번호로 변경 후 실행한다.

## 10. 이행 결과 검증 (oracle)

### 10-1. Python 검증 스크립트

```bash
vi ~/03_verify.py
```

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
import cx_Oracle
import mysql.connector

ORA_USER = "hr"
ORA_PASS = "hr"
ORA_DSN  = "172.31.0.250:1521/orcl"

MY_HOST  = "127.0.0.1"
MY_PORT  = 3306
MY_USER  = "root"
MY_PASS  = "실제_비밀번호_입력"   # ← 수정 필요
MY_DB    = "hr"

TABLES = ["REGIONS","COUNTRIES","LOCATIONS","JOBS","DEPARTMENTS","EMPLOYEES","JOB_HISTORY"]

ora_con = cx_Oracle.connect(ORA_USER, ORA_PASS, ORA_DSN)
ora_cur = ora_con.cursor()

my_con = mysql.connector.connect(
    host=MY_HOST, port=MY_PORT,
    user=MY_USER, password=MY_PASS, database=MY_DB
)
my_cur = my_con.cursor()

print("=" * 50)
print("검증: Oracle vs MySQL 건수 비교")
print("=" * 50)
print("{:<20} {:>8} {:>8} {:>6}".format("테이블","Oracle","MySQL","결과"))
print("-" * 45)

all_ok = True
for t in TABLES:
    ora_cur.execute("SELECT COUNT(*) FROM {}".format(t))
    ora_cnt = ora_cur.fetchone()[0]

    my_cur.execute("SELECT COUNT(*) FROM {}".format(t.lower()))
    my_cnt = my_cur.fetchone()[0]

    ok = "OK" if ora_cnt == my_cnt else "NG"
    if ora_cnt != my_cnt:
        all_ok = False
    print("  {:<18} {:>8} {:>8} {:>6}".format(t, ora_cnt, my_cnt, ok))

print("-" * 45)
if all_ok:
    print("[완료] 모든 테이블 건수 일치")
else:
    print("[경고] 건수 불일치 테이블 있음")

ora_cur.close(); ora_con.close()
my_cur.close();  my_con.close()
```

## 트러블슈팅

| 증상 | 원인 | 해결 |
|------|------|------|
| `ModuleNotFoundError: No module named 'cx_Oracle'` | cx_Oracle 미설치 | root에서 `pip3 install cx_Oracle` |
| `mysql-connector protobuf 오류` | Python 3.6에서 최신 버전 비호환 | `pip3 install mysql-connector-python==8.0.11` |
| `ORA-01034: ORACLE not available` | Oracle DB 미기동 | `sqlplus / as sysdba` → `startup` |
| `ORA-28000: account is locked` | HR 유저 잠김 | `alter user hr identified by hr account unlock` |
| FK 제약 위반 에러 | 이행 순서 문제 | TABLE_ORDER에서 부모 테이블 먼저 이행 |
